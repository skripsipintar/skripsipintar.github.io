[tex] \sqrt{13 {}^{2} - 5 {}^{2} } \\ \sqrt{169 - 25} \\ \sqrt{144} \\ 12 \: cm[/tex]

K = 17 cm + 13 cm + 12 cm + 12 cm

K = 30 cm + 24 cm

K = 54 cm
